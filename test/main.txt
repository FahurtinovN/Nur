def get_min_max(l):
  try:
    sl = {
      'max': 0,
      'min': 0
    }
    max = l[0]
    min = l[0]
    for i in l:
        if i > max:
            max = i
        if i < min:
            min = i
    sl['max'] = int(max)
    sl['min'] = int(min)
    return sl
  except:
    return "Введите нормальный список"

def zamsp(l):
  try:
    for i in range(0, len(l)):
      znac = l[i]
      l[i] = []
      for g in range(1, i+2):
        if znac % g == 0:
          l[i].append(g)
    return l
  except:
    return "Введи нормальный список"

import random

def randsp(s1, s2):
  a = []
  b = random.randint(0, 10)
  c = 0
  a.append(random.randint(s1, s2))
  while c < b:
    k = random.randint(s1, s2)
    if a[c] != k:
      a.append(k)
      c += 1
  return a
